-   请求样例

    ```
    POST /v2/{project_id}/{resource_type}/resource_instances/action
    ```

    -   action为filter时请求体

        ```
        {
          "offset": "100", 
          "limit": "100", 
          "action": "filter", 
          "matches":[
               {
                "key": "resource_name", 
                "value": "resource1"
               }
           ], 
           "not_tags": [
               {
                "key": "key1", 
                "values": ["*value1","value2"]
               }，
               {
                "key": "key2", 
                "values": ["*value21","value22"]
               }
           ], 
           "tags": [
            {
              "key": "key1", 
              "values": ["*value1","value2"]
              }
             ], 
           "tags_any": [
            {
              "key": "key1", 
              "values": ["value1", "value2"]
            }
          ],
           "not_tags_any": [
            {
              "key": "key1", 
              "values": ["value1", "value2"]
            }
          ]
        }
        ```

    -   action为count时请求体

        ```
        {
          "action": "count", 
          "not_tags": [
            {
              "key": "key1", 
              "values": ["value1", "*value2"]
            }
          ], 
          "tags": [
            {
              "key": "key1", 
              "values": ["value1", "value2"]
            }
          ], 
          "tags_any": [
            {
              "key": "key1", 
              "values": [ "value1", "value2"]
            }
          ],
          "not_tags_any": [
            {
              "key": "key1", 
              "values": ["value1", "value2"]
            }
           ],
           "matches":[
           {
                "key": "resource_name", 
                "value": "resouurce"
           }
          ]
        }
        ```
